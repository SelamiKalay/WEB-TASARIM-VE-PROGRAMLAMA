<h2>Convert a string into a function.</h2>
<p id="demo"></p>

<script>
const text = '{"name":"ken", "age":"function() {return 10;}", "city":"Tokyo"}';
const obj = JSON.parse(text);
obj.age = eval("(" + obj.age + ")");
document.getElementById("demo").innerHTML = obj.name + ", " + obj.age(); 
</script>

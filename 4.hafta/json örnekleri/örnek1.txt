<script>

'{"firstName":"Arda","lastName":"Baş" },' +
'{"firstName":"Emo","lastName":"Yılmaz" },' +
'{"firstName":"Şafak","lastName":"Acer" }]}';

obj.employees[2].firstName + " " + obj.employees[2].lastName;
